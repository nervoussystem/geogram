379  bool compute_inverse(matrix_type& result) const {
380  FT val=FT(0.0), val2=FT(0.0);
381  matrix_type tmp = (*this);
382 
383  result.load_identity();
384 
385  for(index_t i = 0; i != DIM; i++) {
386  val = tmp(i, i); /* find pivot */
387  index_t ind = i;
388  for(index_t j = i + 1; j != DIM; j++) {
389  if(fabs(tmp(j, i)) > fabs(val)) {
390  ind = j;
391  val = tmp(j, i);
392  }
393  }
394 
395  if(ind != i) {
396  for(index_t j = 0; j != DIM; j++) {
397  val2 = result(i, j);
398  result(i, j) = result(ind, j);
399  result(ind, j) = val2; /* swap columns */
400  val2 = tmp(i, j);
401  tmp(i, j) = tmp(ind, j);
402  tmp(ind, j) = val2;
403  }
404  }
405 
406  if(val == 0.0) {
407  return false;
408  }
409 
410  for(index_t j = 0; j != DIM; j++) {
411  tmp(i, j) /= val;
412  result(i, j) /= val;
413  }
414 
415  for(index_t j = 0; j != DIM; j++) {
416  if(j == i) {
417  continue; /* eliminate column */
418  }
419  val = tmp(j, i);
420  for(index_t k = 0; k != DIM; k++) {
421  tmp(j, k) -= tmp(i, k) * val;
422  result(j, k) -= result(i, k) * val;
423  }
424  }
425  }
426 
427  return true;
428  }
