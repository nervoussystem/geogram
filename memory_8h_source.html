265  inline void* aligned_malloc(
266  size_t size, size_t alignment = GEO_MEMORY_ALIGNMENT
267  ) {
268 #if defined(GEO_OS_ANDROID)
269  // Alignment not supported under Android.
270  geo_argused(alignment);
271  return malloc(size);
272 #elif defined(GEO_COMPILER_INTEL)
273  return _mm_malloc(size, alignment);
274 #elif defined(GEO_COMPILER_GCC) || defined(GEO_COMPILER_CLANG)
275  void* result;
276  return posix_memalign(&result, alignment, size) == 0
277  ? result : nullptr;
278 #elif defined(GEO_COMPILER_MSVC)
279  return _aligned_malloc(size, alignment);
280 #else
281  geo_argused(alignment);
282  return malloc(size);
283 #endif
284  }

412  inline void* align(void* p) {
413  size_t offset = (
414  GEO_MEMORY_ALIGNMENT -
415  (reinterpret_cast<size_t>(p) & (GEO_MEMORY_ALIGNMENT - 1))
416  ) & (GEO_MEMORY_ALIGNMENT - 1);
417  return reinterpret_cast<char*>(p) + offset;
418  }
