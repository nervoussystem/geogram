660  if(picking_mode_ == MESH_NONE) {

715  void draw_vertex(index_t v) {
716  if(do_animation_) {
717  if(mesh_->vertices.single_precision()) {
718  const GLUPfloat* p =
719  mesh_->vertices.single_precision_point_ptr(v);
720  float t = float(time_);
721  float s = 1.0f - float(time_);
722  glupPrivateVertex3f(
723  s*p[0] + t*p[3],
724  s*p[1] + t*p[4],
725  s*p[2] + t*p[5]
726  );
727  } else {
728  const GLUPdouble* p = mesh_->vertices.point_ptr(v);
729  double s = 1.0 - time_;
730  glupPrivateVertex3d(
731  s*p[0] + time_*p[3],
732  s*p[1] + time_*p[4],
733  s*p[2] + time_*p[5]
734  );
735  }
736  } else {
737  if(mesh_->vertices.single_precision()) {
738  if(mesh_->vertices.dimension() < 3) {
739  glupPrivateVertex2fv(
740  mesh_->vertices.single_precision_point_ptr(v)
741  );
742  } else {
743  glupPrivateVertex3fv(
744  mesh_->vertices.single_precision_point_ptr(v)
745  );
746  }
747  } else {
748  if(mesh_->vertices.dimension() < 3) {
749  glupPrivateVertex2dv(
750  mesh_->vertices.point_ptr(v)
751  );
752  } else {
753  glupPrivateVertex3dv(
754  mesh_->vertices.point_ptr(v)
755  );
756  }
757  }
758  }
